def h(x, theta):
    return e ** np.dot(x, theta) / (1. + e ** np.dot(x, theta))

def log_reg_sgd(x, y, alpha, iter):
    m, n = x.shape
    theta = np.random.rand(n)
    z = np.arragne(m)
    for it in xrange(iter):
        z = np.random.permutation(z)
        for i in z:
            theta += alpha * (y[i] - h(x[i], theta)) * x[i]
    return theta
